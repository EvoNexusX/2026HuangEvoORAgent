### 决策变量
- \(x_{ij}\)：从仓库 \(i\) 运送到港口 \(j\) 的集装箱数量，为非负整数，其中 \(i \in I\)，\(j \in J\)。
- \(y_{ij}\)：从仓库 \(i\) 到港口 \(j\) 所需的卡车数量，为非负整数，其中 \(i \in I\)，\(j \in J\)。

索引集合：  
\(I = \{1,2,3,4,5,6\}\) 表示仓库，对应顺序为：1=Verona，2=Perugia，3=Rome，4=Pescara，5=Taranto，6=Lamezia。  
\(J = \{1,2,3,4,5\}\) 表示港口，对应顺序为：1=Genoa，2=Venice，3=Ancona，4=Naples，5=Bari。

### 目标函数
最小化总运输成本：
\[
\min \sum_{i \in I} \sum_{j \in J} 30 \cdot d_{ij} \cdot y_{ij}
\]
其中 \(d_{ij}\) 为从仓库 \(i\) 到港口 \(j\) 的距离（单位：公里），具体数值由题目给出的距离表确定。

### 约束条件
1. **供应约束**：每个仓库运出的集装箱总数不超过其库存量。
   \[
   \sum_{j \in J} x_{ij} \leq s_i, \quad \forall i \in I
   \]
   其中 \(s_i\) 为仓库 \(i\) 的库存量：\(s_1=10\)，\(s_2=12\)，\(s_3=20\)，\(s_4=24\)，\(s_5=18\)，\(s_6=40\)。

2. **需求约束**：每个港口接收的集装箱总数等于其需求量。
   \[
   \sum_{i \in I} x_{ij} = d_j, \quad \forall j \in J
   \]
   其中 \(d_j\) 为港口 \(j\) 的需求量：\(d_1=20\)，\(d_2=15\)，\(d_3=25\)，\(d_4=33\)，\(d_5=21\)。

3. **卡车容量约束**：每条路线上的集装箱数量不超过卡车运力的两倍。
   \[
   x_{ij} \leq 2 y_{ij}, \quad \forall i \in I, j \in J
   \]

4. **非负整数约束**：
   \[
   x_{ij} \in \mathbb{Z}_{\ge 0}, \quad y_{ij} \in \mathbb{Z}_{\ge 0}, \quad \forall i \in I, j \in J
   \]

### 数学模型
综合以上，得到如下混合整数线性规划模型：

\[
\begin{aligned}
\min \quad & \sum_{i=1}^{6} \sum_{j=1}^{5} 30 \cdot d_{ij} \cdot y_{ij} \\
\text{s.t.} \quad & \sum_{j=1}^{5} x_{ij} \leq s_i, \quad i = 1,\dots,6, \\
& \sum_{i=1}^{6} x_{ij} = d_j, \quad j = 1,\dots,5, \\
& x_{ij} \leq 2 y_{ij}, \quad i = 1,\dots,6, \; j = 1,\dots,5, \\
& x_{ij}, y_{ij} \in \mathbb{Z}_{\ge 0}, \quad i = 1,\dots,6, \; j = 1,\dots,5.
\end{aligned}
\]

其中距离 \(d_{ij}\) 由下表给出（单位：公里）：

| 仓库 \ 港口 | Genoa | Venice | Ancona | Naples | Bari  |
|-------------|--------|---------|---------|---------|-------|
| Verona      | 290    | 115     | 355     | 715     | 810   |
| Perugia     | 380    | 340     | 165     | 380     | 610   |
| Rome        | 505    | 530     | 285     | 220     | 450   |
| Pescara     | 655    | 450     | 155     | 240     | 315   |
| Taranto     | 1010   | 840     | 550     | 305     | 95    |

### 导入模块
import gurobipy as gp
from gurobipy import GRB

### 参数与数据
# 仓库和港口的索引集合
I = range(6)  # 0:Verona, 1:Perugia, 2:Rome, 3:Pescara, 4:Taranto, 5:Lamezia
J = range(5)  # 0:Genoa, 1:Venice, 2:Ancona, 3:Naples, 4:Bari

# 仓库库存量
s = [10, 12, 20, 24, 18, 40]

# 港口需求量
d = [20, 15, 25, 33, 21]

# 距离矩阵（公里）：dist[i][j] 表示从仓库i到港口j的距离
dist = [
    [290, 115, 355, 715, 810],   # Verona
    [380, 340, 165, 380, 610],   # Perugia
    [505, 530, 285, 220, 450],   # Rome
    [655, 450, 155, 240, 315],   # Pescara
    [1010, 840, 550, 305, 95],   # Taranto
    [1072, 1097, 747, 372, 333]  # Lamezia
]

### 模型构建
# 1. 创建模型
m = gp.Model("Transportation")

# 2. 定义决策变量
x = m.addVars(I, J, vtype=GRB.INTEGER, name="x")
y = m.addVars(I, J, vtype=GRB.INTEGER, name="y")

# 3. 设置目标函数：最小化总运输成本
m.setObjective(gp.quicksum(30 * dist[i][j] * y[i, j] for i in I for j in J), GRB.MINIMIZE)

# 4. 添加约束
# 供应约束：每个仓库运出的集装箱总数不超过其库存量
supply_constrs = m.addConstrs((gp.quicksum(x[i, j] for j in J) <= s[i] for i in I), name="supply")

# 需求约束：每个港口接收的集装箱总数等于其需求量
demand_constrs = m.addConstrs((gp.quicksum(x[i, j] for i in I) == d[j] for j in J), name="demand")

# 卡车容量约束：每条路线上的集装箱数量不超过卡车运力的两倍
truck_constrs = m.addConstrs((x[i, j] <= 2 * y[i, j] for i in I for j in J), name="truck")

### 求解与输出
# 1. 求解模型
m.optimize()

# 2. 输出结果
if m.status == GRB.OPTIMAL:
    print(f"最优总成本: {m.objVal:.2f}")
    print("\n运输方案 (x_ij):")
    for i in I:
        for j in J:
            if x[i, j].x > 0:
                print(f"  从仓库{i+1}到港口{j+1}: {int(x[i, j].x)} 个集装箱")
    print("\n卡车使用数量 (y_ij):")
    for i in I:
        for j in J:
            if y[i, j].x > 0:
                print(f"  从仓库{i+1}到港口{j+1}: {int(y[i, j].x)} 辆卡车")
else:
    print("未找到最优解")

   \sum_{j=1}^{4} x_{ij} \leq 1, \quad \forall i = 1,2,3,4,5

### 导入模块
import gurobipy as gp
from gurobipy import GRB

### 参数与数据
# 工人数量
num_workers = 5
# 任务数量
num_tasks = 4
# 成本矩阵 c[i][j] 表示工人i完成任务j的小时数
c = [
    [9, 4, 3, 7],
    [4, 6, 5, 6],
    [5, 4, 7, 5],
    [7, 5, 2, 3],
    [10, 6, 7, 4]
]

### 模型构建
# 1. 创建模型
m = gp.Model("Assignment_Problem")

# 2. 定义决策变量：x[i][j]为二元变量
x = m.addVars(num_workers, num_tasks, vtype=GRB.BINARY, name="x")

# 3. 设置目标函数：最小化总工作时间
m.setObjective(gp.quicksum(c[i][j] * x[i, j] for i in range(num_workers) for j in range(num_tasks)), GRB.MINIMIZE)

# 4. 添加约束
# 每个任务必须由恰好一个工人完成
for j in range(num_tasks):
    m.addConstr(gp.quicksum(x[i, j] for i in range(num_workers)) == 1, name=f"Task_{j}")

# 每个工人最多完成一个任务
for i in range(num_workers):
    m.addConstr(gp.quicksum(x[i, j] for j in range(num_tasks)) <= 1, name=f"Worker_{i}")

### 求解与输出
# 1. 求解模型
m.optimize()

# 2. 输出结果
if m.status == GRB.OPTIMAL:
    print(f"最优总工作时间: {m.ObjVal} 小时")
    print("任务分配情况:")
    for i in range(num_workers):
        for j in range(num_tasks):
            if x[i, j].X > 0.5:  # 由于是二元变量，值接近1即表示被选中
                # 注意：索引从0开始，转换为从1开始以便阅读
                worker_name = f"工人 {i+1}"
                task_name = chr(ord('A') + j)  # 任务A, B, C, D
                print(f"  {worker_name} 被分配给任务 {task_name}，工作时间: {c[i][j]} 小时")
else:
    print("未找到最优解")

import gurobipy as gp
from gurobipy import GRB

### 参数与数据

wood_limit = 890
steel_limit = 500

# 利润系数
profit = [5, 10, 8, 7]  # 对应卡车、飞机、船、火车

max_production = [74, 44, 59, 89]  # 对应卡车、飞机、船、火车

m = gp.Model()

# 2. 定义决策变量
# x: 生产数量（非负整数）

# y: 是否生产（二元变量）
y = m.addVars(4, vtype=GRB.BINARY, name="y")

# 3. 设置目标函数
m.setObjective(gp.quicksum(profit[i] * x[i] for i in range(4)), GRB.MAXIMIZE)

# 4. 添加约束
# 木材资源约束: 12x1 + 20x2 + 15x3 + 10x4 <= 890

# 钢材资源约束: 6x1 + 3x2 + 5x3 + 4x4 <= 500
m.addConstr(6 * x[0] + 3 * x[1] + 5 * x[2] + 4 * x[3] <= steel_limit, "steel_constraint")

# 卡车与火车互斥逻辑: y1 + y4 <= 1
m.addConstr(y[0] + y[3] <= 1, "truck_train_exclusive")

# 船与飞机依赖逻辑: y3 <= y2
m.addConstr(y[2] <= y[1], "ship_depend_on_plane")

# 船的数量不超过火车的数量: x3 <= x4
m.addConstr(x[2] <= x[3], "ship_le_train")

# 生产数量与二元变量关联约束: 
# x1 <= 74*y1, x1 >= y1
# x2 <= 44*y2, x2 >= y2
# x3 <= 59*y3, x3 >= y3
# x4 <= 89*y4, x4 >= y4

    m.addConstr(x[i] <= max_production[i] * y[i], f"upper_bound_{i}")
    m.addConstr(x[i] >= y[i], f"lower_bound_{i}")

m.optimize()

    print("是否生产:")
    for i in range(4):
        print(f"  y[{i+1}] = {y[i].x}")
else:
