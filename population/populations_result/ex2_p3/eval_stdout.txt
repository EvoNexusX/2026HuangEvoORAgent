from gurobipy import GRB

def main():

    Solves the empty container transportation problem using Gurobi.
    The objective is to minimize total transportation cost while satisfying
    supply and demand constraints with truck capacity limitations.
    """
    
    # Create model
    model = gp.Model("EmptyContainerTransportation")
    
    # Data preparation
    # Warehouses and their supply
    warehouses = ['Verona', 'Perugia', 'Rome', 'Pescara', 'Taranto', 'Lamezia']
    supply = {

        'Venice': 15,
        'Ancona': 25,
        'Naples': 33,

    }

    }
    
    # Constants
    C = 30  # Cost per km per truck (euros)

from gurobipy import GRB

def main():
    # 1. 模型初始化
    model = gp.Model("ToyProduction")
    
    # 2. 定义决策变量
    # 整数变量：生产数量
    x1 = model.addVar(vtype=GRB.INTEGER, lb=0, name="Trucks")
    x2 = model.addVar(vtype=GRB.INTEGER, lb=0, name="Airplanes")
    x3 = model.addVar(vtype=GRB.INTEGER, lb=0, name="Boats")
    x4 = model.addVar(vtype=GRB.INTEGER, lb=0, name="Trains")

    y1 = model.addVar(vtype=GRB.BINARY, name="y_Trucks")
    y2 = model.addVar(vtype=GRB.BINARY, name="y_Airplanes")
    y3 = model.addVar(vtype=GRB.BINARY, name="y_Boats")
    y4 = model.addVar(vtype=GRB.BINARY, name="y_Trains")
    
    # 3. 设置目标函数
    model.setObjective(5*x1 + 10*x2 + 8*x3 + 7*x4, GRB.MAXIMIZE)
    
    # 4. 添加约束条件
    # 资源约束
